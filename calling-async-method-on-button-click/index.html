Calling async method on button click

Action that you want to execute:

public async Task<string> ExecuteAsync(string command)
{
    return await Task.Run(() =>
    {
        using (var client = new SshClient(hostname, port, username, password))
        {
            client.Connect();
            return client.CreateCommand(command).Execute();
        }
    });
}

And async button handler:

private async void buttonExecuteCommandAsync_Click(object sender, EventArgs eventArgs)
{
    VAppSSHManager vapp = new VAppSSHManager("hostname.pl", 22, "user", "password");
    String result = await vapp.ExecuteAsync("ls -la");
    Console.Out.WriteLine(result);
}
