1 February 2018 · Docker

NGINX + GHOST + JENKINS etc.

docker-compose.yml

version: "2"
services:
  nginx:
    image: nginx
    container_name: nginx
    restart: always
    links:
      - ghost
      - jenkins
    ports:
      - "80:80"
    volumes:
      - /home/debian/nginx:/etc/nginx

  ghost:
    image: ghost:1.19
    container_name: ghost
    restart: always
    expose:
      - "2368"
    environment:
      - url=http://blog.wapacz.nazwa.pl/
    volumes:
      - /home/debian/ghost/ghost-blog-dump:/var/lib/ghost/content

  jenkins:
    image: jenkins
    container_name: jenkins
    restart: always
    expose:
      - "8080"
      - "50000"
    volumes:
      - /home/debian/jenkins:/var/jenkins_home
